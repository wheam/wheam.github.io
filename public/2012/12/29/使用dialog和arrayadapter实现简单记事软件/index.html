接着拟写Voice类：


01 package sssta.hellovoice;
02
03 public class Voice {
04
05 String name;//声音的描述
06 int time;//声音长度
07 String path=“/mnt/sdcard/seemore/”+name+“.mp3”;//直接使用name作为声音路径
08
09 //构造方法，传入一个string，一个int值，分别赋值给name和time
10 public Voice(String a,int b)
11 {
12 name=a;
13 time=b;
14 }
15 }
