使用MediaRecoder和MediaPlayer实现简单录音播放软件



如标题，讲解看注释，项目已经上传github，https://github.com/wheam/MediaRecorderTry

简单的录音播放软件，实现了一下功能：

录制声音，播放声音，存储声音到SD卡，文件存储异常处理，部分防止错误操作的处理



【主界面，中间的文字用于提示】



【恶意卖萌】

148 private void Record()
149 {
150 //建立实例
151 mediaRecorder=new MediaRecorder();
152 //设置声音获取的方式，此处为麦克风
153 mediaRecorder.setAudioSource(MediaRecorder.AudioSource.MIC);
154 //设置声音的封装格式
155 mediaRecorder.setOutputFormat(MediaRecorder.OutputFormat.THREE_GPP);
156 //设置声音的编码格式
157 mediaRecorder.setAudioEncoder(MediaRecorder.AudioEncoder.AMR_NB);
158 //为path赋值
159
160 //设置声音的保存路径
161 mediaRecorder.setOutputFile(path);
162 //try-catch语句是为了捕捉异常
163 try {
164 //准备和播放，拟写者两句，系统会自动提示捕捉异常
165 mediaRecorder.prepare();
166 mediaRecorder.start();
167 //下面是两个异常，自动添加
168 } catch (IllegalStateException e) {
169 // TODO Auto-generated catch block
170 e.printStackTrace();
171 } catch (IOException e) {
172 // TODO Auto-generated catch block
173 e.printStackTrace();
174 }
175 }

237 //如果程序文件夹不存在，创建一个新的文件夹
238 private void createSDCardDir(){
239 if(Environment.MEDIA_MOUNTED.equals(Environment.getExternalStorageState())){
240 // 创建一个文件夹对象，赋值为外部存储器的目录
241 File sdcardDir =Environment.getExternalStorageDirectory();
242 //得到一个路径，内容是sdcard的文件夹路径和名字
243 String path=sdcardDir.getPath()+“/MediaRecorderTry”;
244 File path1 = new File(path);
245 if (!path1.exists()) {
246 //若不存在，创建目录，可以在应用启动的时候创建
247 path1.mkdirs();
248 }
249 }
250 else{
251 return;
252 }
